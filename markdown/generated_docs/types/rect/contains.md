# contains

>  Check if a point lies within this rectangle, inclusive of its edges.
>  # Examples
>  ```
>  # use bevy_math::Rect;
>  let r = Rect::new(0., 0., 5., 1.); // w=5 h=1
>  assert!(r.contains(r.center()));
>  assert!(r.contains(r.min));
>  assert!(r.contains(r.max));
>  ```
